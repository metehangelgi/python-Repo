		function debounce(fn, delay) {
		  var timer = null;
		  return function () {
			var context = this, args = arguments;
			clearTimeout(timer);
			timer = setTimeout(function () {
			  fn.apply(context, args);
			}, delay);
		  };
		}

				var aj = $('#ajax-search').val();

				if (keywords.length > 2 && aj == "idle") {
					$(".result-info").html('Sonuçlar aranıyor...');
					// $('#ajax-search').val('working');
					$.ajax({
						type: 'post',
						url: aurl,
						data: '_post=search&keywords='+keywords,
						success: function (r) {
							$('body').append(r);
							$('#ajax-search').val('a');

							setTimeout(function() {
								$('#ajax-search').val('idle');
							}, 500);
						}
					});
				}
				else {
